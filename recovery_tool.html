Local Cache Recovery

Run this in your Main App Console (F12) to dump your browser's private offline database.

(async () => {
    console.log("🕵️ Starting Local Cache Extraction...");
    const dbName = "firestore/[DEFAULT]/faiora-24f4a/main";
    const request = indexedDB.open(dbName);
    request.onsuccess = (e) => {
        const db = e.target.result;
        console.log("✅ IndexedDB Opened. Searching documents...");
        const tx = db.transaction(["remoteDocuments"], "readonly");
        const store = tx.objectStore("remoteDocuments");
        const getReq = store.getAll();
        getReq.onsuccess = () => {
            const docs = getReq.result;
            const matches = docs.filter(d => JSON.stringify(d).includes("Hard Talks"));
            console.log(`🔎 Found ${matches.length} potential matches for "Hard Talks" in Local Cache!`);
            matches.forEach(m => console.log("MATCH FOUND:", m));
            if(matches.length === 0) console.log("❌ No local matches found. Data may have been overwritten.");
            else alert("SUCCESS: Check your browser console for matches!");
        };
    };
    request.onerror = () => console.error("Could not open IndexedDB.");
})();
